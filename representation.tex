\section{Représentation}
\subsection{Isomorphismes accidentels}

\begin{frame}[fragile]
  \begin{block}{Isomorphismes accidentels}
    $$Spin(1) = O(1)$$
    $$Spin(2) = U(1) = O(2)$$
    $$Spin(3) = Sp(1) = SU(2)$$
    $$Spin(4) = Sp(1) × Sp(1)$$
    $$Spin(5) = Sp(2)$$
    $$Spin(6) = SU(4)$$
  \end{block}
\end{frame}
