\section{Interprétation physique}

\begin{frame}
  \begin{block}{Le spin, une rotation ?}
    un objet possède un spin $s\,$ s'il est invariant sous une
    rotation d'angle $\frac{2\pi}{s}\,$.\\[1ex]
    $s\,$ peut être entier ou demi-entier.\\[1ex]
    En mécanique quantique, le spin est un opérateur vectoriel
    hermitien.
  \end{block}
  \begin{block}{Fermions: Spin demi-entier}
    \begin{itemize}
      \item Statistique de Fermi-Dirac.
      \item Le principe d'exclusion de Pauli.
    \end{itemize}
  \end{block}
  \begin{block}{Bosons: Spin entier}
    \begin{itemize}
      \item Statistique de Bose-Einstein.
      \item Condensat de Bose-Einstein $\Rightarrow$ Superfluide.
    \end{itemize}
  \end{block}  
\end{frame}
